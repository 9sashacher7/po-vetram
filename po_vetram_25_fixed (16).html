Google Sheets — архив заявок
Как подключить Google Sheets — 4 шага:

1. Откройте sheets.google.com → создайте новую таблицу «По Ветрам»
2. В меню: Расширения → Apps Script → вставьте код ниже → сохраните
3. Нажмите «Развернуть» → «Создать развёртывание» → тип «Веб-приложение» → доступ «Все»
4. Скопируйте URL развёртывания и вставьте в поле выше
Код для Google Apps Script
// ═══════════════════════════════════════════
// По Ветрам — Google Apps Script
// Версия с полной синхронизацией данных
// ═══════════════════════════════════════════

function doGet(e) {
  var action = e && e.parameter && e.parameter.action;
  
  // Вернуть все данные (заявки + агенты)
  if (action === 'getData') {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    
    // Заявки
    var ordersSheet = ss.getSheetByName('Заявки') || ss.getActiveSheet();
    var ordersData = ordersSheet.getDataRange().getValues();
    var orders = [];
    if (ordersData.length > 1) {
      for (var i = 1; i < ordersData.length; i++) {
        var r = ordersData[i];
        if (!r[1]) continue; // пропустить пустые строки
        orders.push({
          _row    : i + 1,
          date    : r[0] ? r[0].toString() : '',
          id      : r[1] ? r[1].toString() : '',
          name    : r[2] ? r[2].toString() : '',
          phone   : r[3] ? r[3].toString() : '',
          service : r[4] ? r[4].toString() : '',
          dateWalk: r[5] ? r[5].toString() : '',
          slot    : r[6] ? r[6].toString() : '',
          persons : r[7] ? r[7].toString() : '',
          addons  : r[8] ? r[8].toString() : '',
          amount  : r[9] ? Number(r[9]) : 0,
          agent   : r[10] ? r[10].toString() : '',
          income  : r[11] ? Number(r[11]) : 0,
          status  : r[12] ? r[12].toString() : 'Новая',
          comment : r[13] ? r[13].toString() : '',
          pct     : r[14] ? Number(r[14]) : 10,
        });
      }
    }
    
    // Агенты
    var agentsSheet = ss.getSheetByName('Агенты');
    var agents = [];
    if (agentsSheet) {
      var agentsData = agentsSheet.getDataRange().getValues();
      if (agentsData.length > 1) {
        for (var j = 1; j < agentsData.length; j++) {
          var a = agentsData[j];
          if (!a[0]) continue;
          agents.push({
            name    : a[0] ? a[0].toString() : '',
            code    : a[1] ? a[1].toString().toUpperCase() : '',
            pct     : a[2] ? Number(a[2]) : 10,
            phone   : a[3] ? a[3].toString() : '',
            tg      : a[4] ? a[4].toString() : '',
            initials: a[5] ? a[5].toString() : '',
          });
        }
      }
    }
    
    return ContentService
      .createTextOutput(JSON.stringify({ status:'ok', orders:orders, agents:agents }))
      .setMimeType(ContentService.MimeType.JSON);
  }
  
  return ContentService
    .createTextOutput(JSON.stringify({status:'active'}))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  var sheet = SpreadsheetApp.getActiveSpreadsheet();
  var data = JSON.parse(e.postData.contents);
  
  // ── Обновить статус ──
  if (data._action === 'updateStatus') {
    var ordersSheet = sheet.getSheetByName('Заявки') || sheet.getActiveSheet();
    var rows = ordersSheet.getDataRange().getValues();
    for (var i = 1; i < rows.length; i++) {
      if (rows[i][1] == data.id) {
        ordersSheet.getRange(i+1, 13).setValue(data.status);
        break;
      }
    }
    return ContentService
      .createTextOutput(JSON.stringify({status:'updated'}))
      .setMimeType(ContentService.MimeType.JSON);
  }
  
  // ── Сохранить агента ──
  if (data._action === 'saveAgent') {
    var agentsSheet = sheet.getSheetByName('Агенты');
    if (!agentsSheet) {
      agentsSheet = sheet.insertSheet('Агенты');
      agentsSheet.appendRow(['Имя','Код','%','Телефон','Telegram','Инициалы']);
    }
    // Проверить не существует ли уже
    var aRows = agentsSheet.getDataRange().getValues();
    for (var k = 1; k < aRows.length; k++) {
      if (aRows[k][1] === data.code) {
        // Обновить процент если изменился
        agentsSheet.getRange(k+1, 3).setValue(data.pct);
        return ContentService
          .createTextOutput(JSON.stringify({status:'updated'}))
          .setMimeType(ContentService.MimeType.JSON);
      }
    }
    agentsSheet.appendRow([data.name, data.code, data.pct, data.phone||'', data.tg||'', data.initials||'']);
    return ContentService
      .createTextOutput(JSON.stringify({status:'saved'}))
      .setMimeType(ContentService.MimeType.JSON);
  }
  
  // ── Новая заявка ──
  var ordersSheet2 = sheet.getSheetByName('Заявки') || sheet.getActiveSheet();
  if (ordersSheet2.getLastRow() === 0) {
    ordersSheet2.appendRow([
      'Дата','№ Заявки','Имя','Телефон','Услуга',
      'Дата прогулки','Слот','Кол-во','Доп. услуги',
      'Сумма, ₽','Агент','Доход агента, ₽','Статус','Комментарий','% агента'
    ]);
    ordersSheet2.getRange('J:J').setNumberFormat('0');
    ordersSheet2.getRange('L:L').setNumberFormat('0');
    ordersSheet2.getRange('O:O').setNumberFormat('0');
    ordersSheet2.getRange('M:M').setNumberFormat('@');
  }
  
  var persons = parseInt(data.persons) || 0;
  var amount  = parseInt(data.amount)  || 0;
  var pct     = parseInt(data.pct)     || 10;
  var earn    = Math.round(amount * pct / 100);
  
  ordersSheet2.appendRow([
    new Date().toLocaleString('ru'),
    data.id, data.name, data.phone, data.service,
    data.date, data.slot, persons, data.addons,
    amount, data.agent, earn, 'Новая',
    data.comment || '', pct
  ]);
  
  return ContentService
    .createTextOutput(JSON.stringify({status:'ok'}))
    .setMimeType(ContentService.MimeType.JSON);
}
